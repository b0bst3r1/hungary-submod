    }
}

defined_text = { 
    name = HUN_unrest_impact_hungary
    
    text = { 
        trigger = {
            OR = {
            check_variable = { HUN_Hungarian_Unrest > 50 }
            check_variable = { HUN_Hungarian_Unrest = 50 }
            }
        }
        localization_key = "HUN_unrest_impact_hungary_high"
    }
    
    text = { 
        trigger = {
            OR = {
            check_variable = { HUN_Hungarian_Unrest > 25 }
            check_variable = { HUN_Hungarian_Unrest = 25 }
            }
        }
        localization_key = "HUN_unrest_impact_hungary_medium"
    }
    
    text = { 
        trigger = {
            OR = {
            check_variable = { HUN_Hungarian_Unrest < 25 }
            check_variable = { HUN_Hungarian_Unrest = 0 }
            }
        }
        localization_key = "HUN_unrest_impact_hungary_low"
    }
}
