   open_the_ROM_window = {
       window_name = "Romanian_Legionaries_Window"
       parent_window_token = politics_tab
       context_type = player_context
 
       visible = {
            tag = ROM
       }
 
        effects = {
            ROM_open_click = {
                if = {
                    limit = {	
                        NOT = {
                            has_country_flag = ROM_window_open
                        }
                    }
                    set_country_flag  = ROM_window_open
                }
            }
        }
        triggers = {
            ROM_open_click_enabled = {
                ROOT = { 
                    OR = { 
                        has_government = ni 
                        has_government = ds 
                        has_government = rc 
                    }
                }
            }
        }
    }
 }
